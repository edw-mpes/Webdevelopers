    <script>
        const app = {
            state: {
                cart: JSON.parse(localStorage.getItem('wp_cart')) || null,
                orders: JSON.parse(localStorage.getItem('wp_orders')) || [],
                paymentMethod: null
            },

            init() {
                this.updateUI();
                this.initPayPal();
            },

            showPage(pageId) {
                document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
                document.getElementById(pageId + 'Page').classList.add('active');
                document.querySelectorAll('.nav-link').forEach(l => l.classList.remove('active'));
                window.scrollTo(0, 0);
            },

            addToCart(name, price) {
                this.state.cart = { name, price };
                localStorage.setItem('wp_cart', JSON.stringify(this.state.cart));
                this.updateUI();
                this.showPage('checkout');
                this.notify("Προστέθηκε στο καλάθι!", "success");
            },

            updateUI() {
                const cartCount = document.getElementById('cartCount');
                const cartDisplay = document.getElementById('cartDisplay');
                const cartTotal = document.getElementById('cartTotal');

                if (this.state.cart) {
                    cartCount.textContent = '1';
                    cartDisplay.textContent = `Πακέτο: ${this.state.cart.name}`;
                    cartTotal.textContent = `€${this.state.cart.price}`;
                } else {
                    cartCount.textContent = '0';
                    cartDisplay.textContent = 'Το καλάθι είναι άδειο.';
                    cartTotal.textContent = '€0';
                }
            },

            setPayment(method) {
                this.state.paymentMethod = method;
                document.querySelectorAll('.payment-method').forEach(m => m.classList.remove('selected'));
                document.getElementById('pay-' + method).classList.add('selected');

                const paypalContainer = document.getElementById('paypal-button-container');
                const manualBtn = document.getElementById('manualPayBtn');

                if (method === 'paypal') {
                    paypalContainer.style.display = 'block';
                    manualBtn.style.display = 'none';
                } else {
                    paypalContainer.style.display = 'none';
                    manualBtn.style.display = 'block';
                }
            },

            initPayPal() {
                if (!window.paypal) return;
                
                paypal.Buttons({
                    createOrder: (data, actions) => {
                        if (!this.state.cart) return alert("Το καλάθι είναι άδειο!");
                        return actions.order.create({
                            purchase_units: [{ amount: { value: this.state.cart.price.toString() } }]
                        });
                    },
                    onApprove: (data, actions) => {
                        return actions.order.capture().then(details => {
                            this.completeOrder("PayPal Success: " + details.id);
                        });
                    }
                }).render('#paypal-button-container');
            },

            processManualPayment() {
                const name = document.getElementById('custName').value;
                const email = document.getElementById('custEmail').value;
                
                if (!name || !email) return this.notify("Συμπληρώστε τα στοιχεία σας!", "error");
                
                // INSERT STRIPE / CREDIT CARD LOGIC HERE
                this.notify("Επεξεργασία Κάρτας...", "warning");
                setTimeout(() => this.completeOrder("Card Payment"), 2000);
            },

            completeOrder(paymentInfo) {
                const orderId = 'ORD-' + Math.random().toString(36).substr(2, 6).toUpperCase();
                const newOrder = {
                    id: orderId,
                    customer: document.getElementById('custName').value,
                    package: this.state.cart.name,
                    price: this.state.cart.price,
                    date: new Date().toLocaleDateString(),
                    status: 'Σε εξέλιξη'
                };

                this.state.orders.push(newOrder);
                localStorage.setItem('wp_orders', JSON.stringify(this.state.orders));
                
                this.state.cart = null;
                localStorage.removeItem('wp_cart');
                this.updateUI();

                alert("Η παραγγελία ολοκληρώθηκε! Κωδικός: " + orderId);
                this.showPage('home');
            },

            trackOrder() {
                const id = document.getElementById('trackInput').value.trim().toUpperCase();
                const order = this.state.orders.find(o => o.id === id);
                const res = document.getElementById('trackResult');

                if (order) {
                    res.innerHTML = `<div style="background: var(--card); padding: 2rem; border-radius: 15px; border-left: 5px solid var(--accent);">
                        <h4>Παραγγελία ${order.id}</h4>
                        <p>Κατάσταση: <span style="color: var(--warning)">${order.status}</span></p>
                        <p>Πακέτο: ${order.package}</p>
                    </div>`;
                } else {
                    res.textContent = "Δεν βρέθηκε η παραγγελία.";
                }
            },

            notify(msg, type) {
                const box = document.getElementById('toastBox');
                const t = document.createElement('div');
                t.className = 'toast';
                t.textContent = msg;
                t.style.borderLeft = `5px solid var(--${type})`;
                box.appendChild(t);
                setTimeout(() => t.remove(), 3000);
            }
        };

        app.init();
    </script>
