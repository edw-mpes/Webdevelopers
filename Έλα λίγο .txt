        function formatCardNumber(input) {

        function formatExpiry(input) {
            let value = input.value.replace(/\D/g, '');
            if (value.length >= 2) {
                value = value.substring(0, 2) + '/' + value.substring(2, 4);
            }

            document.getElementById('cardExpiryPreview').textContent = value || 'MM/YY';
        }

        function validateCardNumber(number) {
            const cleaned = number.replace(/\s/g, '');
            if (cleaned.length < 13 || cleaned.length > 19) return false;
            let sum = 0, isEven = false;
            for (let i = cleaned.length - 1; i >= 0; i--) {
                let digit = parseInt(cleaned[i], 10);
                if (isEven) { digit *= 2; if (digit > 9) digit -= 9; }
                sum += digit; isEven = !isEven;
            }
            return sum % 10 === 0;
        }

        function validateExpiry(expiry) {
            const parts = expiry.split('/');
            if (parts.length !== 2) return false;
            const month = parseInt(parts[0], 10), year = parseInt('20' + parts[1], 10);
            if (month < 1 || month > 12) return false;
            const now = new Date(), currentYear = now.getFullYear(), currentMonth = now.getMonth() + 1;
            return !(year < currentYear || (year === currentYear && month < currentMonth));
        }

        function validateCVV(cvv) { return /^\d{3,4}$/.test(cvv); }

        function validateAndProcess() {
            if (!validateName() || !validateEmail()) return;
            if (cart.length === 0) { showToast('Παρακαλώ επιλέξτε ένα πακέτο.', 'error'); return; }
            if (!selectedPayment) { showToast('Παρακαλώ επιλέξτε τρόπο πληρωμής.', 'error'); return; }
        }

        function processPaysafe() {
            if (!validateName() || !validateEmail() || cart.length === 0) return;
            const code = document.getElementById('paysafeCode').value;
            if (!/^\d{16}$/.test(code)) {
                document.getElementById('paysafeError').classList.add('show');
                return;
            }
            document.getElementById('processingModal').classList.add('active');
            setTimeout(() => { document.getElementById('processingModal').classList.remove('active'); completeOrder('Paysafe'); }, 2000);
        }

        function processCard() {
            if (!validateName() || !validateEmail() || cart.length === 0) return;
            const num = document.getElementById('cardNumber').value, exp = document.getElementById('cardExpiry').value, cvv = document.getElementById('cardCvv').value;
            let err = false;
            if (!validateCardNumber(num)) { document.getElementById('cardNumberError').classList.add('show'); err = true; }
            if (!validateExpiry(exp)) { document.getElementById('cardExpiryError').classList.add('show'); err = true; }
            if (!validateCVV(cvv)) { document.getElementById('cardCvvError').classList.add('show'); err = true; }
            if (err) return;
            document.getElementById('processingModal').classList.add('active');
            setTimeout(() => { document.getElementById('processingModal').classList.remove('active'); completeOrder(selectedPayment.toUpperCase()); }, 2000);
        }

        function completeOrder(method) {
            const orderId = 'ORD-' + Date.now().toString().slice(-6);
            const name = document.getElementById('customerName').value;
            const email = document.getElementById('customerEmail').value;
            const order = { id: orderId, package: cart[0].name, price: cart[0].price, customer: name, email: email, payment: method, date: new Date().toISOString() };
            orders.push(order);
            document.getElementById('successOrderId').textContent = orderId;
            document.getElementById('successPackage').textContent = order.package;
            document.getElementById('successCustomer').textContent = name;
            document.getElementById('successEmail').textContent = email;
            document.getElementById('successPayment').textContent = method;
            document.getElementById('successTotal').textContent = '€' + order.price;
            cart = []; document.getElementById('cartCount').textContent = '0';
            showPage('success');
        }

        function trackOrder() {
            const input = document.getElementById('trackingInput').value.trim();
            const result = document.getElementById('trackingResult');
            const order = orders.find(o => o.id === input);
            if (order) {
                result.innerHTML = `<div class="order-details-card"><h4>Παραγγελία ${order.id}</h4><div class="detail-row"><span>Πακέτο:</span><strong>${order.package}</strong></div><div class="detail-row"><span>Κατάσταση:</span><strong style="color: var(--success);">Ολοκληρωμένη</strong></div><div class="detail-row"><span>Σύνολο:</span><strong style="color: var(--accent);">€${order.price}</strong></div></div>`;
            } else {
                result.innerHTML = `<div class="order-details-card"><p style="color: var(--error); text-align: center;">Δεν βρέθηκε παραγγελία.</p></div>`;
            }
        }

        function handleNewsletter(e) { e.preventDefault(); showToast('Ευχαριστούμε!', 'success'); e.target.reset(); }

        function showToast(msg, type = 'info') {
            const container = document.getElementById('toastContainer');
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            toast.innerHTML = `<span>${type === 'success' ? '✓' : 'ℹ'}</span><span>${msg}</span>`;
            container.appendChild(toast);
            setTimeout(() => { toast.remove(); }, 3000);
        }

            const header = document.querySelector('header');
            window.scrollY > 50 ? header.classList.add('scrolled') : header.classList.remove('scrolled');
        });
